        # Parse "ColName:type" pairs

        $colTypes  = @{}
        if ($rawStr) {
            foreach ($pair in ($rawStr -split ',\s*')) {

                if ($parts.Count -ge 2) {
                    $colName = $parts[0].Trim()
                    $colType = $parts[1].Trim()
                    $availCols.Add($colName)
                    $colTypes[$colName] = $colType
                }

            ComputerField    = $cfg.ComputerField
            UserField        = $cfg.UserField
            TimeField        = $cfg.TimeField
